3   /**
4    * Luokka pitää kirjaa arvattavasta sanasta.
5    * @author sepkim
6    */
7   public class Sana {
8       
9       private String sana;
10      private boolean vinkkiKayttamatta;
11      private boolean[] arvatutKirjaimet;
12      
13      /**
14       * Konstruktori alustaa tarvittavat muuttujat.
15       * @param sana 
16       */
17      public Sana(String sana) {
18          this.sana = sana;
19          this.vinkkiKayttamatta = true;
20          arvatutKirjaimet = new boolean[sana.length()];
21      }
22      
23      /**
24       * Arpoo uuden sanan.
25       * @param arpoja Arpoja jota käytetään arpomiseen.
26       */
27      public void uusiSana(SanaArpoja arpoja) {
28          sana = arpoja.arvoSana();
29          this.vinkkiKayttamatta = true;
30      }
31      
32      /**
33       * Käyttää vinkin.
34       */
35      public void kaytaVinkki() {
36          this.vinkkiKayttamatta = false;
37      }
38      
39      /**
40       * Tarkistaa vinkin tilan.
41       * @return Palauttaa true jos vinkki on käyttämättä.
42       */
43      public boolean vinkkiKayttamatta() {
44          return vinkkiKayttamatta;
45      }
46      
47      public boolean[] getArvatutKirjaimet() {
48          return arvatutKirjaimet;
49      }
50      
51      /**
52       * Palauttaa merkkijonon jossa arvattavasta sanasta on näkyvillä vain ne 
53       * kirjaimet, jotka on jo arvattu.
54       * @return Palauttaa sanan siten että arvaamattomat kirjaimet ovat piilossa.
55       */
56      public String sanaPiilotetuillaKirjaimilla() {
57          String palautettava = "";
58          int i = 0;
59          
60          for (char kirjain : sana.toCharArray()) {     //lisää arvatut kirjaimet palautettavaan merkkijonoon,
61              if (arvatutKirjaimet[i] == true) {        //muiden kohdalle alaviiva
62                  palautettava += kirjain;
63              } else {
64                  palautettava += "_";
65              }
66              
67              if (i < sana.length() - 1) {
68                  palautettava += ".";
69              }
70              i++;
71          }
72          return palautettava;
73      }
74      
75      /**
76       * Arvaa kirjaimen.
77       * @return Palauttaa true jos arvaus oli oikein.
78       * @param kirjain Kirjain jota pelaaja arvaa.
79       */
80      public boolean arvaa(char kirjain) {
81          int i = 0;
82          boolean oikeinArvattu = false;
83          
84          for (char c : sana.toCharArray()) {
85              if (c == kirjain) {
86                  arvatutKirjaimet[i] = true;
87                  oikeinArvattu = true;
88              }
89              
90              i++;
91          }
92          return oikeinArvattu;
93      }
94      
95      /**
96       * Tarkistaa onko kaikki sanan kirjaimet jo arvattu.
97       * @return palauttaa true jos kaikki kirjaimet on arvattu.
98       */
99      public boolean kaikkiKirjaimetArvattu() {
100         boolean voitto = true;
101         for (boolean b : arvatutKirjaimet) {
102             if (!b) {
103                 voitto = false;
104             }
105         }
106         
107         return voitto;
108     }
109 
110     @Override
111     public String toString() {
112         return sana;
113     }
114 
115     public String getSana() {
116         return sana;
117     }
118     
119     /**
120      * Asettaa kaikki kirjaimet näkyviksi.
121      */
122     public void naytaKokoSana() {
123         for (int i = 0; i < arvatutKirjaimet.length; i++) {
124             arvatutKirjaimet[i] = true;
125         }
126     }
127     
128     
129 }
