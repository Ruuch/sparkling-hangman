19       * Avaa tiedoston ja lukee sanat listaan.

22      public SanaArpoja() throws Exception {

33      public String arvoSana() {

36      }                                                       //merkkijonon joka on listassa numeron osoittamalla sijainnilla

38      /**

40       */
41      private void lueSanatListaan()  {
42  
43          String merkkijono = "";
44  
45          try {
46              while (lukija.hasNextLine()) {
47                  merkkijono = lukija.nextLine();
48                  sanat.add(merkkijono);
49              }
50  
51              lukija.close();
52          } catch (Exception e) {
53              System.out.println("Tiedostoa lukiessa sattui virhe!");
54          }
55  
56      }
